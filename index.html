Conversor de Horas para Minutos e Segundos

Copie o código e teste no DartPad! DartPad.dev

                        class Horas {
                            int _horario = 0, _minutos = 0, _segundos = 0;
                          
                            // Setter
                            set horario(int novoHorario) => _horario = novoHorario;
                          
                            // Getter
                            int get horario => _horario;
                            int get minutos => _minutos;
                            int get segundos => _segundos;
                          
                            void converterEmMinutos() {
                              this._minutos = this._horario * 60;
                            }
                          
                            void converterEmSegundos() {
                              this._segundos = this._horario * 60 * 60;
                            }
                          }
                          
                          void main() {
                            var horas = Horas();
                          
                            horas.horario = 3;
                            horas.converterEmMinutos();
                            horas.converterEmSegundos();
                          
                            print("Hora: ${horas.horario}, Em minutos: ${horas.minutos}");
                            print("Hora: ${horas.horario}, Em segundos: ${horas.segundos}");
                          }

                          // Getter
                            double get idadeAproximada => _idadeAproximada;
                          
                          void calculoDaIdade(){
                            this._idadeAproximada = this._anoAtual - this._diaDoNascimento;
                           }
                          }
                          
                          void main(){

                            idade.anoAtual = 2025;

Calculadora de Valor de Produto

                        class Produto {
                            double _valorProduto = 0;
                            int _quantidadeProduto = 0;
                          
                            //Getter
                            double get valorProduto => _valorProduto;
                          
                            //Setter
                            set valorProduto(double novoValorProduto) => _valorProduto = novoValorProduto;
                          
                            //Getter
                            int get quantidadeProduto => _quantidadeProduto;
                          
                            //Setter
                            set quantidadeProduto(int novoQuantidadeProduto) =>
                                _quantidadeProduto = novoQuantidadeProduto;
                          
                            //Calculo do produto em função da quantidade
                            double calculo() {
                              return (this._valorProduto * this._quantidadeProduto);
                            }
                          }
                          
                          void main() {
                            var produto = Produto();
                            double resultado = 0;
                          
                            produto._valorProduto = 100;
                            produto._quantidadeProduto = 2;
                            resultado = produto.calculo();
                          
                            print("Valor do produto: ${produto._valorProduto}");
                            print("Quant. do produto: ${produto._quantidadeProduto}");
                            print("Total a pagar: ${resultado}");
                          }

Calculadora de Salário com Descontos e Acréscimos

Copie o código e teste no DartPad! DartPad.dev

                        class SalarioFinal {
                            double _salarioBase = 2000;
                            double _valorHora = 9;
                            int _horasExtra = 0;
                            int _faltas = 0;
                          
                            // Getter
                            double get salarioBase => _salarioBase;
                          
                            // Getter
                            double get valorHora => _valorHora;
                          
                            // Getter
                            int get horasExtra => _horasExtra;
                          
                            // Setter
                            set horasExtra(int novoHorasExtra) => _horasExtra = novoHorasExtra;
                          
                            // Getter
                            int get faltas => _faltas;
                          
                            // Setter
                            set faltas(int novoFaltas) => _faltas = novoFaltas;
                          
                            // Calculos
                            // Salário Base
                            double salarioBruto() {
                              return _salarioBase;
                            }
                          
                            // Calculo do acrescimo
                            double calculoAcrescimo() {
                              return _horasExtra * _valorHora;
                            }
                          
                            // Calculo do Desconto
                            double calculoDesconto() {
                              return _faltas * _valorHora;
                            }
                          
                            // Calculo do Salário Liquido
                            double salarioLiquido() {
                              return _salarioBase + calculoAcrescimo() - calculoDesconto();
                            }
                          }
                          
                          void main() {
                            var salarioFinal = SalarioFinal();
                          
                            salarioFinal.horasExtra = 10;
                            salarioFinal.faltas = 5;
                          
                            print("Salário Bruto: ${salarioFinal.salarioBruto()}");
                            print("Salário Liquido: ${salarioFinal.salarioLiquido()}");
                            print("Valor Desconto: ${salarioFinal.calculoDesconto()}");
                            print("Valor Acréscimo: ${salarioFinal.calculoAcrescimo()}");
                          }
